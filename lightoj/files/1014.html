	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	int cas=1;
	while (t--){
		cout<<"Case "<<cas++<<": ";
        ll p,l;
        cin>>p>>l;
        p=p-l;
        vector < ll > ans;
        for (ll i=1;i*i<=p;i++){
            if (p%i == 0){
                if (i > l)
                    ans.pb(i);
                if (p/i > l && i*i != p)
                    ans.pb(p/i);
            }
        }
        if (ans.size() >= 1 ){
            sort(ans.begin(),ans.end());
            for (int i=0;i<ans.size()-1;i++){
                cout<<ans[i]<<" ";
            }
            cout<<ans[ans.size()-1]<<endl;
        }
        else
            cout<<"impossible"<<endl;
	}
	return 0;
}
