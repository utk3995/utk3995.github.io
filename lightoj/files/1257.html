Problem Name : Problem 1257 - Farthest Nodes in a Tree (II)


Problem Tags : Breadth First Search/Depth First Search, Dynamic Programming

Problem Statement :

Given a tree (a connected graph with no cycles), you have to find the cost to go to the farthest node from each node. The edges of the tree are weighted and undirected.


Code : 

/* Logic : The logic is the basis of the fact that if we select
an arbitrary vertex and find the farthest vertex from it, say A,
then A is a point on the diameter. Now lets select a vertex C on
the diameter AB. We say that the farthest point from C will either
be A or B. If it was any other vertex D, then our assumption that
AB is the diameter is wrong. Take any example and demonstrate.*/


#include <bits/stdc++.h>
using namespace std;
#define pii pair<int,int>
#define vi vector<int>
#define vii vector < pii >
#define si set<int>
#define ll long long int
#define pb push_back
#define mp make_pair
#define fr first
#define se second
#define MOD 1000003
#define MAX 10100

int n;
vector < pii > g[30100];
ll dis[30100];
int deg[30100];
int vis[30100];
ll fc;
int fi;

void clea(int n)
{
    for (int i=0;i<=n+10;i++){
        g[i].clear();
        dis[i]=0;
        deg[i]=0;
        vis[i]=0;
    }
    fc=0;
    fi=0;
}

ll dfs1(int u,ll cd)
{
    //cout<<u<<endl;
    ll ma = 0;
    vis[u]=1;
    for (int i=0;i<g[u].size();i++){
        int v = g[u][i].fr;
        ll w = g[u][i].se;
        if (vis[v]==1)
            continue;
        ma = max(ma, w + dfs1(v,cd+w));
    }
    dis[u]=max(dis[u],max(ma,cd));
    if (cd > fc){
        fc=cd;
        fi=u;
    }
    return ma;
}

ll dfs2(int u,ll cd)
{
    //cout<<u<<endl;
    ll ma = 0;
    vis[u]=2;
    for (int i=0;i<g[u].size();i++){
        int v = g[u][i].fr;
        ll w = g[u][i].se;
        if (vis[v]==2)
            continue;
        ma = max(ma, w + dfs2(v,cd+w));
    }
    dis[u]=max(dis[u],max(ma,cd));
    return ma;
}


int main ()
{
    //cout << setprecision(7) << fixed;
	#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);
	#endif
	//ios_base::sync_with_stdio(false);
	int t;
	scanf("%d",&t);
	int cas=1;
	while (t--){
		printf("Case %d:\n",cas++);
        scanf("%d",&n);
        clea(n);
        for (int i=0;i<n-1;i++){
            int u,v,w;
            scanf("%d %d %d",&u,&v,&w);
            g[u].pb(mp(v,w));
            g[v].pb(mp(u,w));
            deg[u]++;
            deg[v]++;
        }
        dfs1(0,0);
        int ffi=fi;
        fc=fi=0;
        for (int i=0;i<n;i++){
            dis[i]=vis[i]=0;
        }
        dfs1(ffi,0);
        dfs2(fi,0);
        for (int i=0;i<n;i++)
            printf("%lld\n",dis[i]);
	}
	return 0;
}
