int dp[120][120];
int n;
int a[120];

int func(int low,int hi)
{
    if (low == hi)
        return 1;
    if (low > hi)
        return 0;
    if (dp[low][hi] != -1)
        return dp[low][hi];
    int ret = func(low+1,hi)+1;
    for (int i=low+1;i<=hi;i++){
        if (a[i]==a[low]){
            ret = min(ret,func(low,i-1)+func(i+1,hi));
        }
    }
    return (dp[low][hi] = ret);
}
