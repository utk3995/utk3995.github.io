Problem Name : Problem 1110 - An Easy LCS


Problem Tags : Dynamic Programming

Problem Statement :

LCS means 'Longest Common Subsequence' that means two non-empty strings are given; you have to find the Longest Common Subsequence between them. Since there can be many solutions, you have to print the one which is the lexicographically smallest. Lexicographical order means dictionary order. For example, 'abc' comes before 'abd' but 'aaz' comes before 'abc'.


Code : 

#include <bits/stdc++.h>
using namespace std;
#define pii pair<int,int>
#define vi vector<int>
#define vii vector < pii >
#define si set<int>
#define ll long long int
#define pb push_back
#define mp make_pair
#define fr first
#define se second
#define MOD 1000003
#define MAX 10100

int main ()
{
    //cout << setprecision(7) << fixed;
	#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);
	#endif
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	int cas=1;
	while (t--){
		cout<<"Case "<<cas++<<": ";
        string a,b;
        cin>>a>>b;
        int n=a.size();
        int m=b.size();
        int dp[n+1][m+1];
        string dp1[n+1][m+1];
        for (int i=0;i<=n;i++){
            for (int j=0;j<=m;j++){
                if (i==0 || j==0){
                    dp[i][j]=0;
                    dp1[i][j]="";
                }
                else{
                    if (dp[i-1][j] > dp[i][j-1]){
                        dp1[i][j]=dp1[i-1][j];
                        dp[i][j]=dp[i-1][j];
                    }
                    else if(dp[i-1][j] < dp[i][j-1]){
                        dp1[i][j]=dp1[i][j-1];
                        dp[i][j]=dp[i][j-1];
                    }
                    else{
                        dp1[i][j]=min(dp1[i][j-1],dp1[i-1][j]);
                        dp[i][j]=dp[i-1][j];
                    }
                    if (a[i-1]==b[j-1]){
                        dp[i][j]=max(dp[i][j],dp[i-1][j-1]+1);
                        if (dp1[i-1][j-1].size()+1 > dp1[i][j].size())
                            dp1[i][j]=dp1[i-1][j-1]+a[i-1];
                        else{
                            if (dp1[i-1][j-1]+a[i-1] < dp1[i][j])
                                dp1[i][j] = dp1[i-1][j-1] + a[i-1];
                        }
                    }
                }
            }
        }
        string ans = dp1[n][m];
        if (ans=="")
            ans=":(";
        cout<<ans<<endl;
	}
	return 0;
}
