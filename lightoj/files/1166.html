int main ()
{

	ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    int cas=1;
    while(t--){
        int n;
        cin>>n;
        int a[n];
        int b[n];
        for (int i=0;i<n;i++){
            cin>>a[i];
            b[i]=a[i];
        }
        sort(b,b+n);
        int co=0;
        for (int i=0;i<n;i++){
            if (a[i] != i+1){
                for (int j=i+1;j<n;j++){
                    if (a[j]==i+1){
                        co++;
                        swap(a[i],a[j]);
                    }
                }
            }
        }
        cout<<"Case "<<cas++<<": "<<co<<endl;
    }
	return 0;
}
