Problem Name : Problem 1043 - Triangle Partitioning


Problem Tags : Basic Geometry, Binary Search/Bisection

Problem Statement :

See the picture below.

You are given AB, AC and BC. DE is parallel to BC. You are also given the area ratio between ADE and BDEC. You have to find the value of AD.


Code : 

#include <bits/stdc++.h>
using namespace std;
#define pii pair<int,int>
#define vi vector<int>
#define vii vector < pii >
#define si set<int>
#define ll long long int
#define pb push_back
#define mp make_pair
#define fr first
#define se second
#define MOD 1000003
#define MAX 10100
#define eps 1e-7

bool check(double ab,double ac,double bc,double ad,double r)
{
    double ae = (ad/ab*ac);
    double de = (ad/ab*bc);
    double s1 = (ad+ae+de)/2.0;
    double s2 = (ab+bc+ac)/2.0;
    double abc = sqrt(s2*(s2-ab)*(s2-bc)*(s2-ac));
    double ade = sqrt(s1*(s1-ad)*(s1-ae)*(s1-de));
    double bdec = abc-ade;
    if (((ade/bdec) - r) >= eps)
        return true;
    else
        return false;
}

int main ()
{
    cout << setprecision(8) << fixed;
	#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);
	#endif
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	int cas=1;
	while (t--){
		cout<<"Case "<<cas++<<": ";
        double ab,ac,bc,r;
        cin>>ab>>ac>>bc>>r;
        double low=0;
        double high=ab;
        double ans=-1;
        for (int i=0;i<64;i++){
            double mid=(low+high)/2;
            if (check(ab,ac,bc,mid,r)){
                high=mid;
                ans=mid;
            }
            else{
                low=mid;
            }
        }
        cout<<ans<<endl;
	}
	return 0;
}
