int ans[5000005];
ull fin[5000005];

void precompu()
{
    for (int i=0;i<5000005;i++)
        ans[i]=i;
    for (int i=2;i<5000005;i++){
        if (ans[i]==i){
            ans[i]=i-1;
            for (int j=2;j*i<5000005;j++){
                ans[i*j] = (ans[i*j]/i)*(i-1);
            }
        }
    }
    fin[1]=0;
    for (int i=2;i<5000005;i++){
        ull kk = ans[i];
        fin[i]=fin[i-1]+kk*kk;
    }
}
